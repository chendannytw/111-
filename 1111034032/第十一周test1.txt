#include <stdio.h>
#define SIZE 10 

void test1(int a[],int b[],int size);

int main(void){
	int a[SIZE]={2,6,4,8,10,12,89,68,45,37};
    int s[SIZE]={66,67,66,70,73,74,75,62,32}; 
	printf("氣泡排序前");
	int i;
	for(i = 0 ; i<SIZE ; ++i)
	printf("%4d",a[i]);//印分數 
	printf("\n");
	printf("          ");
	for(i=0 ; i<SIZE ; ++i)
	printf("%4d",s[i]);//印結果 
	
	printf("\n氣泡排序後");
	for(i=0;i<SIZE;++i)
	printf("%4d",a[i]);//印分數
	printf("\n");
	printf("          ");
    test1(a,s,SIZE);//呼叫副程式 
    for(i=0 ; i<SIZE ; ++i)
	printf("%4d",s[i]);//印結果 
	
}

void test1(int a[],int s[],int size){
	int pass,hold,i;
	for (pass=1 ; pass<SIZE ; ++pass){
		//執行幾回合 
		for(i=0 ; i<SIZE-pass ; ++i){//每回合裡面做幾次 
			if(s[i]>s[i+1]){
				hold = s[i];//定義hold用於暫存較大值 
				s[i] = s[i + 1];
				s[i + 1] = hold;
				
				hold = a[i];//定義hold用於暫存較大值 
				a[i] = a[i + 1];
				a[i + 1] = hold;
			}
		}
	}
}
